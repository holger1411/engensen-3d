```ts
/** Persistenter Missions-Fortschritt (Freischaltung des Endlevels). */
const KEY = "horde30938.completed";
export const VILLAGE_IDS = ["engensen", "thoense", "lahberg", "wettmar", "oldhorst"];

export function loadCompleted(): Set<string> {
  try {
    if (typeof localStorage === "undefined") return new Set();
    return new Set(JSON.parse(localStorage.getItem(KEY) || "[]") as string[]);
  } catch {
    return new Set();
  }
}

export function markCompleted(id: string): void {
  if (typeof localStorage === "undefined") return;
  const s = loadCompleted();
  s.add(id);
  localStorage.setItem(KEY, JSON.stringify([...s]));
}

/** Endlevel `grossburgwedel` ist erst frei, wenn alle Dörfer gewonnen sind. */
export function isUnlocked(id: string, completed: Set<string>): boolean {
  if (id !== "grossburgwedel") return true;
  return VILLAGE_IDS.every((v) => completed.has(v));
}
```
